

function diffArray(arr1, arr2) {
  var newArr = [];
  // Same, same; but different.
  let mergedArr = arr1.concat(arr2);
  mergedArr.forEach(function(item){
    if(arr1.indexOf(item) === -1 || arr2.indexOf(item) === -1){
      newArr.push(item);
    }
  });
    
  return newArr;
}

console.log(diffArray([1, 2, 3, 5], [1, 2, 3, 4, 5]));
